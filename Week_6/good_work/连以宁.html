import numpy as np
from sklearn.model_selection import train_test_split

n_class, n_len, width, height = 11, 5, 28, 28#why 11?

def generate_dataset(X, y):
    X_len = X.shape[0]
    
    X_gen = np.zeros((X_len, height, width*n_len, 1), dtype=np.uint8)
    y_gen = [np.zeros((X_len, n_class), dtype=np.uint8)for i in range(n_len)]#for i in range(n_len)#5,48000,11 #list of numpy array
    # TODO: 1~5
    for n in range(X_len):
        np.random.seed(n)
        random_num = np.random.choice(n_len-1, 1)
        choice = np.random.choice(X_len,random_num+1)
        for i in range(5):
            if i <= random_num:
                this_picture = X[choice[i]]
                label = y[choice[i]]
                X_gen[n,:,i*width:(i+1)*width,0] = this_picture
                y_gen[i][n][label] = 1
            else:
                y_gen[i][n][10] = 1
    return X_gen, y_gen

def acc(X,y,model):
    y_pred = model.predict(X)
    acc_all = []
    for j in range(n_len):
        acc = 0
        for i in range(y[0].shape[0]):
            predict_value = np.argmax(y_pred[j][i])
            real_value = np.argmax(y[j][i])
            if predict_value == real_value:
                acc += 1
        acc = float(acc) / y[0].shape[0]
        acc = round(acc,4)
        acc_all.append(acc)
    return acc_all
